#include<stdio.h>
#include<conio.h>
void main()
{
int c,first,last,middle,n,search, array[100];
clrscr();
printf("Enter number of elements");
scanf("%d", &n);
for(c=0;c<n;c++)
{
printf("Enter element");
scanf("%d",&array[c]);
}
printf("Enter value to find");
scanf("%d", &search);
first=0;
last=n-1;
middle=(first+last)/2;
while(first<=last)
{
if(array[middle]<search)
{
first=middle+1;
}
else if(array[middle]==search)
{
printf("%d found at location %d\n", search, middle); break;
}
else
{
last=middle-1;
}
middle=(first+last)/2;
}
if(first>last)
{
printf("%d Not found in list\n", search);
}
getch();
}
